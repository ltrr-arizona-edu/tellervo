Login:

<tellervo xmlns="http://www.tellervo.org/schema/1.0" xmlns:tridas="http://www.tridas.org/1.3">

     <authenticate username="yourusername" password="yourpassword" />
  </request>
</tellervo>

  <request type="read">
     <entity type="object" id="e709dc8a-289e-11e4-add1-b7a4fdd7faae" />
  </request>
</tellervo>

  <request type="search">
     <searchParams returnObject="measurementSeries" limit="5">
        <param name="objectid" operator="=" value="136a70a6-566b-546b-a3ae-c48cb046e4cd" />
     </searchParams>
  </request>
</tellervo>
